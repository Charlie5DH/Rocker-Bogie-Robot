1function [dist_max,path_x,path_y] = planificadorBellmanFord(mapMatrix,Re,xini,yini,xfin,yfin)
  2
  3% Este algoritmo determina la trayectoria minima entre un punto inicial y
  4% todos los restantes puntos del entorno
  5% 
  6% el mapa tiene una resolucion de 0.5m lo que significa que la distancia
  7% entre estados es de 0.5m con los sig 4 movimientos:
  8% (i = i+1, j = j) (i = i-1, j = j) (i = i, j = j-1)   (i = i, j = j+1)
  9% el costo de un estado a otro es 0.5m
 10% 
 11% cuando se avanza con un movimiento diagonal(ang = 45) la distancia entre
 12% estados sera sqr(2*Re^2)->costo del movimiento. Esto es con los
 13% movimientos m1,m3,m4,m6
 14% se asume inicio en i=2,j=2
 15
 16[large,wide] = size(mapMatrix);
 17weightsMatrix = mapMatrix;                     %Conjunto de todos los estados
 18n = large*wide;
 19costoMov1 = Re;                                %distancia horizontal
 20costoMov2 = sqrt(2*(Re^2));                    %distancia diagonal
 21
 22for i = 1: 1:large                       %inicializando pesos con infinito
 23    for j = 1: 1: wide
 24        weightsMatrix(i,j) = inf;
 25    end
 26end 
 27Parent_col = weightsMatrix;                 %matriz que almacena las coordenadas de las columnas de los padres
 28Parent_row = weightsMatrix;                 %matriz que almacena las coordenadas de las filas de los padres
 29
 30source = [yini xini];
 31weightsMatrix(source(1),source(2)) = 0;           %coordenadas de inicio
 32
 33for k = 0: 1: n-1
 34    for i = 2: 1: large -1
 35        for j = 2: 1: wide -1
 36            if( mapMatrix(i, j+1) ==1 && (weightsMatrix(i, j+1) > weightsMatrix(i,j) + costoMov1))
 37                weightsMatrix(i, j+1) = weightsMatrix(i,j) + costoMov1;
 38                Parent_col(i,j+1) = j;
 39                Parent_row(i,j+1) = i;
 40            end
 41            if( mapMatrix(i, j-1) ==1 && weightsMatrix(i, j-1) > weightsMatrix(i,j) + costoMov1)
 42                weightsMatrix(i, j-1) = weightsMatrix(i,j) + costoMov1;
 43                Parent_col(i, j-1) = j;
 44                Parent_row(i, j-1) = i;
 45            end
 46            if( mapMatrix(i+1,j) ==1 && weightsMatrix(i+1, j) > weightsMatrix(i,j) + costoMov1)
 47                weightsMatrix(i+1, j) = weightsMatrix(i,j) + costoMov1;
 48                Parent_col(i+1, j) = j;
 49                Parent_row(i+1, j) = i;
 50            end
 51            if( mapMatrix(i-1, j) ==1 && weightsMatrix(i-1, j) > weightsMatrix(i,j) + costoMov1)
 52                weightsMatrix(i-1, j) = weightsMatrix(i,j) + costoMov1;
 53                Parent_col(i-1, j) = j;
 54                Parent_row(i-1, j) = i;
 55            end
 56            if( mapMatrix(i+1, j+1) ==1 && weightsMatrix(i+1, j+1) > weightsMatrix(i,j) + costoMov2)
 57                weightsMatrix(i+1, j+1) = weightsMatrix(i,j) + costoMov2;
 58                Parent_col(i+1, j+1) = j;
 59                Parent_row(i+1, j+1) = i;
 60            end
 61            if( mapMatrix(i-1, j+1) ==1 && weightsMatrix(i-1, j+1) > weightsMatrix(i,j) + costoMov2)
 62                weightsMatrix(i-1, j+1) = weightsMatrix(i,j) + costoMov2;
 63                Parent_col(i-1, j+1) = j;
 64                Parent_row(i-1, j+1) = i;
 65            end
 66            if( mapMatrix(i+1, j-1) ==1 && weightsMatrix(i+1, j-1) > weightsMatrix(i,j) + costoMov2)
 67                weightsMatrix(i+1, j-1) = weightsMatrix(i,j) + costoMov2;
 68                Parent_col(i+1, j-1) = j;
 69                Parent_row(i+1, j-1) = i;
 70            end
 71            if( mapMatrix(i-1, j-1) ==1 && weightsMatrix(i-1, j-1) > weightsMatrix(i,j) + costoMov2)
 72                weightsMatrix(i-1, j-1) = weightsMatrix(i,j) + costoMov2;
 73                Parent_col(i-1, j-1) = j;
 74                Parent_row(i-1, j-1) = i;
 75            end
 76        end
 77    end
 78end
 79
 80dist_max = weightsMatrix(yfin,xfin);                            %distancia del recorrido
 81
 82path_x(1) = xfin;
 83path_y(1) = yfin;
 84C = 2;
 85i = yfin;                                                       %coordenadas fila pto final
 86j = xfin;                                                       %coordenadas col pto final
 87fin = 0;
 88while(fin == 0)                         %conformacion del camino
 89    if(i == inf)                        %no hay camino posible
 90        if (j == inf)
 91            fin = 1;
 92            path_x(:) = 0;
 93            path_y(:) = 0;
 94        end
 95    else 
 96    path_x(C) = Parent_col(i,j);
 97    path_y(C) = Parent_row(i,j);
 98    auxi = i;
 99    auxj = j;
100    i = Parent_row(auxi,auxj);
101    j = Parent_col(auxi,auxj);
102    C = C+1;
103    if(i == yini)                      %si se llega al inicio del camino
104        if(j == xini)
105            fin = 1;
106        end
107    end
108    end
109end
110
111% path_x(C) = Parent_col(i,j);
112% path_y(C) = Parent_row(i,j);
113
114% spy(mapMatrix);
115% hold on;
116% p = plot(path_x,path_y,'k');
117% p.LineWidth = 1.5;
118% grid on;
119% xticks(0:1:large/Re);
120% yticks(0:1:wide/Re);
121% axis([0 large/Re 0 wide/Re]);
122
123end
